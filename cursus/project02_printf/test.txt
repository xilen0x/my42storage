
int ft_printf(char const *fmt, ...)
{
	va_list	args;
	char	*s;
	int		d;
	char	c;
	int		i;

	i = 0;
	va_start(args, fmt);
	while (*fmt)
	{
		if (*fmt == '%')
		{
			fmt++;
			if (*fmt == 's')
			{
				s = va_arg(args, char *);
				while (s[0] != '\0')
				{
					write(1, &*s, 1);
					s++;
					i++;
				}
			}
			else if (*fmt == 'd')
			{
				d = va_arg(args, int);
				ft_putnbr(d);
			}
			else if (*fmt == 'c')
			{
				c = va_arg(args, int);
				ft_putchar((char)c);
			}
			else if (*fmt == 'p') {
                void *ptr = va_arg(args, void *);
                printf("Pointer: %p\n", ptr);
			else
			{
				printf("Formato no v√°lido\n");
			}
		}
		else
			write(1, &*fmt, 1);
		fmt++;
	}
	va_end(args);
	return (0);
}
